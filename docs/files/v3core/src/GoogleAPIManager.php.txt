<?php

namespace eprocess360\v3core;

/**
 * Class GoogleAPIManager
 * Provides access to a connection to the Google Service API
 * @package eprocess360\v3core
 */
class GoogleAPIManager
{
    private static $client;
    private static $cred;

    /**
     * Return a Google API client object, initializing the connection if
     * needed.
     * @return \Google_Client
     * @throws SysVar\Exception\SysVarException
     */
    public static function getAPIClient() {
        global $pool;
        if(!is_object(GoogleAPIManager::$client)) {
            self::$client = new \Google_Client();
            self::$client->setApplicationName($pool->SysVar->get('GoogleAppName'));
            self::$client->setApplicationName($pool->SysVar->get('GoogleAPIProject'));
            self::$client->setDeveloperKey($pool->SysVar->get('GOOGLE_API_KEY'));
        }

        if(!is_object(self::$cred)) {
            $key = base64_decode($pool->SysVar->get('GoogleOAuthServiceKey'));

            self::$cred = new \Google_Auth_AssertionCredentials(
                $pool->SysVar->get('serviceClientEmail'),
                array('https://www.googleapis.com/auth/devstorage.full_control'),
                $key
            );
        }

        self::$client->setAssertionCredentials(self::$cred);
        if (self::$client->getAuth()->isAccessTokenExpired()) {
            self::$client->getAuth()->refreshTokenWithAssertion(self::$cred);
        }

        return self::$client;
    }
}
